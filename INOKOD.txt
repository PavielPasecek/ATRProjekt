#include <Keypad.h>
#include <Wire.h>
#include <LiquidCrystal_I2C.h>
#include <EEPROM.h>

//DISPLEEJ
LiquidCrystal_I2C lcd(0x27,16,2);
unsigned long DisplejPredMillis = 0;

//Keypad
const byte radky = 4;
const byte sloupce = 4;
char keys [radky][sloupce] = {
  {'1','2','3','A'},
  {'4','5','6','B'},
  {'7','8','9','C'},
  {'*','0','#','D'}
};

byte pinyRadek[radky] = {9,8,7,6};
byte pinySloupec[sloupce] = {5,4,3,2};

Keypad myKeypad = Keypad(makeKeymap(keys),pinyRadek,pinySloupec,radky,sloupce);
String inputUzivatele;

//RGB Diodka
enum Barvy {
  CERV,
  ZEL,
  MOD
};

byte DiodaR = 22;
byte DiodaG = 24;
byte DiodaB = 26;


//Senzory
byte Trig1 = 28;
byte Echo1 = 30;
byte Trig2 = 32;
byte Echo2 = 34;

//stavy
enum Stavy {
  POPLACH,
  ODEMCENO,
  ZAMCENO
};

// všechny ostatní promněnný
Stavy momentalniStav = ZAMCENO;
String heslo = "5555"; // 255 % 10 = 5, takze v zakladnim stavu 5555


// Metody (funkce > metoda)
float PrectiSenzor1() {
  digitalWrite(Trig1, LOW);
  delayMicroseconds(2);
  digitalWrite(Trig1, HIGH);
  delayMicroseconds(10);
  digitalWrite(Trig1, LOW);
  int cas = pulseIn(Echo1, HIGH);
  return cas * 0.034 / 2; // prevede na centaky
};

float PrectiSenzor2() {

  delayMicroseconds(10);
  digitalWrite(Trig2, LOW);
  int cas = pulseIn(Echo2, HIGH);
  return cas * 0.034 / 2; // prevede na centaky
};

void PrectiEEPROM()
{
  String vypis = "";
  vypis = (String)(EEPROM.read(0) % 10) + (String)(EEPROM.read(1) % 10) + (String)(EEPROM.read(2) % 10) +(String)(EEPROM.read(3) % 10);
  heslo = vypis;
}

void ZapisEEPROM(String zapis)
{
  for (byte i = 0; i < 4; i++)
  {
    EEPROM.update(i,(int)zapis[i]);
  }
}

void BarvickaNaLedce(Barvy barva)
{
  if (barva == CERV)
  {
    digitalWrite(DiodaR, HIGH); // bude blik
    digitalWrite(DiodaG, LOW);
    digitalWrite(DiodaB, LOW);
  }
  else if (barva == ZEL)
  {
    digitalWrite(DiodaR, LOW);
    digitalWrite(DiodaG, HIGH); // ZELENA BARVA
    digitalWrite(DiodaB, LOW);
  }
  else
  {
    digitalWrite(DiodaR, LOW);
    digitalWrite(DiodaG, LOW); // MODRA BARVA
    digitalWrite(DiodaB, HIGH);
  }
}

void PrepisDisplej()
{
  unsigned long DisplejPoMillis = millis();
  if (DisplejPoMillis - DisplejPredMillis >= 300)  // Kazdych 0.3 vterin se prepise displej
  {
      DisplejPredMillis = DisplejPoMillis;
      lcd.clear();
      lcd.setCursor(0,0);
      lcd.println("Status:" + (String)momentalniStav);
      //lcd.println((String)(myKeypad.getKey()));
  }
}

void setup() {
  lcd.init();
  lcd.backlight();
  Serial.begin(9600);

  pinMode(Trig1, OUTPUT);
  pinMode(Echo1, INPUT);
  pinMode(Trig2, OUTPUT);
  pinMode(Echo2, INPUT);

  pinMode(DiodaR, OUTPUT);
  pinMode(DiodaG, OUTPUT);
  pinMode(DiodaB, OUTPUT);

  PrectiEEPROM();
}

void loop()
{
  BarvickaNaLedce(MOD);
  PrepisDisplej();
  Serial.println("Heslo: " + (String)heslo);
}
